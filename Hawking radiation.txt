# ── Jackson Rung Mapping ──────────────────────────────────────────────────────

def jackson_energy(n):
“”“E_n = ℏ · ω · φⁿ”””

def temp_to_jackson_rung(T):
“”“Find n such that k_B·T = ℏ·ω·φⁿ → n = log(k_B·T/ℏω) / log(φ)”””
E = k_B * T
if E <= 0:
return None
n = np.log(E / (hbar * omega)) / np.log(phi)
return n
